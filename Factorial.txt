package com.Krishna;
import java.util.Scanner;
class Factorial {
	public static void main(String[] args) {
		// Implement your code here 
		Scanner in =new Scanner(System.in);
		int n = in.nextInt();
		int ans = fact(n);
		System.out.println(ans);
		
	}
	static int fact(int n) {
		int fact =1;
		for(int i =1;i<=n;i++){
		    fact*=i;
		}
		return fact;
	}
}
